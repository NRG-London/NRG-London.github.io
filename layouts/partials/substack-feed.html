{{- /* Newsletter items come from data/substack.json (committed to the repo), refreshed by scripts/fetch-thumbnails.ps1. Substack returns 403 to the build-time RSS fetch from CI, so the site reads this committed data rather than fetching live. Returns a slice of dicts: {title, url, summary, date, ts}. */ -}} {{- $items := slice -}} {{- range site.Data.substack -}} {{- $d := time.AsTime .date -}} {{- $items = $items | append (dict "title" .title "url" .url "summary" .summary "date" $d "ts" $d.Unix) -}} {{- end -}} {{- return $items -}}
